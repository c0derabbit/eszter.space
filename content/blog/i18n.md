use-i18n.js
```js
export default function useI18n() {
  const i18n = { hu, en };
  const defaultLang = 'hu';
  const [lang, setLang] = useState(defaultLang);

  function t(string) {
    const keys = string.split('.');

    try {
      let value = i18n[lang];
      while (typeof value !== 'string') value = value[keys.shift()];

      return value;
    } catch (error) {
      return string;
    }
  }

  return { t, lang, setLang };
}
```
